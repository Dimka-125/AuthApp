worker_processes 1;

events {
    worker_connections 1024;
}

http {
    include       mime.types;
    default_type  application/octet-stream;
    sendfile        on;
    keepalive_timeout  65;

    server {
        listen       80;
        server_name  localhost;

location = / {
            return 302 /signin;
        }

        # Обслуживаем форму по адресу /signin
        location = /signin {
            root   C:/ngnix/nginx-1.28.0/html;  # Путь к папке с signin.html
            index  signin.html;
            try_files /signin.html =404;
        }

        # Все остальные запросы проксируем на Spring Boot (порт 8080)
        location / {
            proxy_pass http://localhost:8080;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        }
    }
}
